4пёЏвѓЈ Example Code to Create Folders (JavaScript)

Using Node.js

const fs = require("fs");

const folders = [
 "controllers",
 "models",
 "routes",
 "services",
 "config"
];

folders.forEach(folder => {
 fs.mkdirSync(folder, { recursive: true });
});

console.log("Folders created successfully");
